CREATE TABLE PRODUCT 
   (	PRODUCT_ID NUMBER PRIMARY KEY, 
	PRODUCT_NAME VARCHAR2(200), 
	PRODUCT_PRICE NUMBER, 
	PRODUCT_QTY NUMBER
) ;


CREATE TABLE PRODUCT_COUNTRY 
   (	SEQ_ID NUMBER(5) PRIMARY KEY, 
	PRODUCT_ID_FK NUMBER(5), 
	COUNTRY_NAME VARCHAR2(40), 
	PRODUCT_MODEL NUMBER(5), 
	PRODUCT_COUNTRY_QTY NUMBER(5), 
	CONSTRAINT FK1 FOREIGN KEY (PRODUCT_ID_FK) REFERENCES PRODUCT (PRODUCT_ID)
);


CREATE TABLE USERS 
   (	USR_ID NUMBER PRIMARY KEY , 
	USR_FULL_NAME VARCHAR2(200), 
	EMAIL VARCHAR2(200), 
	USR_NAME VARCHAR2(200), 
	USR_PASSWORD VARCHAR2(200)
);
